# AI驱动内容代理系统 - 部署运维指南

## 概述

本文档详细描述了AI驱动内容代理系统在Cloudflare平台上的部署、配置、监控和运维流程。

## 技术栈

### 核心平台
- **Cloudflare Workers**: 无服务器计算平台
- **Cloudflare D1**: SQLite数据库服务
- **Cloudflare KV**: 键值存储服务
- **Cloudflare R2**: 对象存储服务
- **Cloudflare Vectorize**: 向量搜索服务

### 开发工具
- **Wrangler CLI**: Cloudflare开发工具
- **GitHub Actions**: CI/CD流水线
- **Docker**: 容器化部署
- **Terraform**: 基础设施即代码

```javascript
// src/routes/health.js
export async function healthCheck(c) {
  const checks = {
    timestamp: new Date().toISOString(),
    status: 'healthy',
    version: process.env.VERSION || 'unknown',
    environment: c.env.ENVIRONMENT || 'unknown',
    checks: {
      database: 'unknown',
      cache: 'unknown',
      storage: 'unknown',
      ai_service: 'unknown'
    }
  };

  try {
    // 数据库健康检查
    const dbResult = await c.env.DB.prepare('SELECT 1').first();
    checks.checks.database = dbResult ? 'healthy' : 'unhealthy';
  } catch (error) {
    checks.checks.database = 'unhealthy';
    checks.status = 'degraded';
  }

  try {
    // 缓存健康检查
    await c.env.CACHE.put('health_check', 'ok', { expirationTtl: 60 });
    const cacheResult = await c.env.CACHE.get('health_check');
    checks.checks.cache = cacheResult === 'ok' ? 'healthy' : 'unhealthy';
  } catch (error) {
    checks.checks.cache = 'unhealthy';
    checks.status = 'degraded';
  }

  try {
    // 存储健康检查
    await c.env.STORAGE.put('health_check.txt', 'ok');
    checks.checks.storage = 'healthy';
  } catch (error) {
    checks.checks.storage = 'unhealthy';
    checks.status = 'degraded';
  }

  try {
    // AI服务健康检查
    const response = await fetch(`${c.env.DIFY_BASE_URL}/health`, {
      headers: {
        'Authorization': `Bearer ${c.env.DIFY_API_KEY}`
      }
    });
    checks.checks.ai_service = response.ok ? 'healthy' : 'unhealthy';
  } catch (error) {
    checks.checks.ai_service = 'unhealthy';
    checks.status = 'degraded';
  }

  const statusCode = checks.status === 'healthy' ? 200 : 503;
  return c.json(checks, statusCode);
}
```

```javascript
// src/middleware/metrics.js
export function metricsMiddleware() {
  return async (c, next) => {
    const startTime = Date.now();
    const method = c.req.method;
    const path = c.req.path;
    
    try {
      await next();
      
      const duration = Date.now() - startTime;
      const status = c.res.status;
      
      // 记录性能指标
      await recordMetric(c, {
        metric: 'request_duration',
        value: duration,
        tags: { method, path, status }
      });
      
      await recordMetric(c, {
        metric: 'request_count',
        value: 1,
        tags: { method, path, status }
      });
      
    } catch (error) {
      const duration = Date.now() - startTime;
      
      await recordMetric(c, {
        metric: 'request_duration',
        value: duration,
        tags: { method, path, status: 'error' }
      });
      
      await recordMetric(c, {
        metric: 'error_count',
        value: 1,
        tags: { method, path, error: error.name }
      });
      
      throw error;
    }
  };
}

async function recordMetric(c, metric) {
  try {
    // 发送到Cloudflare Analytics Engine
    await c.env.ANALYTICS.writeDataPoint({
      blobs: [metric.metric],
      doubles: [metric.value],
      indexes: [JSON.stringify(metric.tags)]
    });
  } catch (error) {
    console.error('Failed to record metric:', error);
  }
}
```

```javascript
// src/routes/metrics.js
export async function getMetrics(c) {
  const timeRange = c.req.query('range') || '1h';
  const metric = c.req.query('metric') || 'request_duration';
  
  try {
    // 从Analytics Engine查询指标
    const query = `
      SELECT 
        toStartOfInterval(timestamp, INTERVAL 5 MINUTE) as time,
        avg(doubles[1]) as avg_value,
        max(doubles[1]) as max_value,
        count() as count
      FROM analytics_dataset
      WHERE 
        blobs[1] = '${metric}'
        AND timestamp >= now() - INTERVAL ${timeRange}
      GROUP BY time
      ORDER BY time
    `;
    
    const result = await c.env.ANALYTICS.query(query);
    
    return c.json({
      metric,
      timeRange,
      data: result.data
    });
  } catch (error) {
    return c.json({ error: 'Failed to fetch metrics' }, 500);
  }
}
```

```javascript
// src/utils/logger.js
export class Logger {
  constructor(context = {}) {
    this.context = context;
  }

  log(level, message, data = {}) {
    const logEntry = {
      timestamp: new Date().toISOString(),
      level,
      message,
      context: this.context,
      data,
      requestId: this.context.requestId
    };

    console.log(JSON.stringify(logEntry));
  }

  info(message, data) {
    this.log('info', message, data);
  }

  warn(message, data) {
    this.log('warn', message, data);
  }

  error(message, error) {
    this.log('error', message, {
      error: error.message,
      stack: error.stack,
      name: error.name
    });
  }

  debug(message, data) {
    if (process.env.LOG_LEVEL === 'debug') {
      this.log('debug', message, data);
    }
  }
}

export function createLogger(c) {
  return new Logger({
    requestId: c.get('requestId'),
    method: c.req.method,
    path: c.req.path,
    userAgent: c.req.header('user-agent')
  });
}
```

```javascript
import { createLogger } from '../utils/logger.js';
import { v4 as uuidv4 } from 'uuid';

export function loggingMiddleware() {
  return async (c, next) => {
    const requestId = uuidv4();
    c.set('requestId', requestId);
    
    const logger = createLogger(c);
    c.set('logger', logger);
    
    const startTime = Date.now();
    
    logger.info('Request started', {
      headers: Object.fromEntries(c.req.headers.entries()),
      query: c.req.query()
    });
    
    try {
      await next();
      
      const duration = Date.now() - startTime;
      logger.info('Request completed', {
        status: c.res.status,
        duration
      });
    } catch (error) {
      const duration = Date.now() - startTime;
      logger.error('Request failed', error);
      logger.info('Request completed with error', {
        status: 500,
        duration
      });
      throw error;
    }
  };
}
```

```javascript
// src/utils/alerts.js
export class AlertManager {
  constructor(env) {
    this.env = env;
  }

  async checkAlerts() {
    const alerts = [];
    
    // 检查错误率
    const errorRate = await this.getErrorRate('5m');
    if (errorRate > 0.05) { // 5%错误率阈值
      alerts.push({
        type: 'high_error_rate',
        severity: 'critical',
        message: `Error rate is ${(errorRate * 100).toFixed(2)}%`,
        value: errorRate
      });
    }
    
    // 检查响应时间
    const avgResponseTime = await this.getAverageResponseTime('5m');
    if (avgResponseTime > 2000) { // 2秒阈值
      alerts.push({
        type: 'high_response_time',
        severity: 'warning',
        message: `Average response time is ${avgResponseTime}ms`,
        value: avgResponseTime
      });
    }
    
    // 检查数据库连接
    const dbHealth = await this.checkDatabaseHealth();
    if (!dbHealth) {
      alerts.push({
        type: 'database_unhealthy',
        severity: 'critical',
        message: 'Database health check failed'
      });
    }
    
    // 发送告警
    for (const alert of alerts) {
      await this.sendAlert(alert);
    }
    
    return alerts;
  }

  async sendAlert(alert) {
    try {
      // 发送到Slack
      if (this.env.SLACK_WEBHOOK_URL) {
        await this.sendSlackAlert(alert);
      }
      
      // 发送邮件
      if (this.env.EMAIL_WEBHOOK_URL) {
        await this.sendEmailAlert(alert);
      }
      
      // 记录告警
      await this.logAlert(alert);
    } catch (error) {
      console.error('Failed to send alert:', error);
    }
  }

  async sendSlackAlert(alert) {
    const color = alert.severity === 'critical' ? 'danger' : 'warning';
    const emoji = alert.severity === 'critical' ? '🚨' : '⚠️';
    
    const payload = {
      attachments: [{
        color,
        title: `${emoji} ${alert.type.toUpperCase()}`,
        text: alert.message,
        fields: [
          {
            title: 'Severity',
            value: alert.severity,
            short: true
          },
          {
            title: 'Time',
            value: new Date().toISOString(),
            short: true
          }
        ]
      }]
    };
    
    await fetch(this.env.SLACK_WEBHOOK_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    });
  }
}
```

```javascript
      }
    }
    
    await next();
  };
```

```javascript
// src/utils/encryption.js
import { webcrypto } from 'crypto';

export class EncryptionService {
  constructor(secretKey) {
    this.secretKey = secretKey;
  }

  async encrypt(data) {
    const encoder = new TextEncoder();
    const dataBuffer = encoder.encode(JSON.stringify(data));
    
    const key = await webcrypto.subtle.importKey(
      'raw',
      encoder.encode(this.secretKey.slice(0, 32)),
      { name: 'AES-GCM' },
      false,
      ['encrypt']
    );
    
    const iv = webcrypto.getRandomValues(new Uint8Array(12));
    const encrypted = await webcrypto.subtle.encrypt(
      { name: 'AES-GCM', iv },
      key,
      dataBuffer
    );
    
    return {
      data: Array.from(new Uint8Array(encrypted)),
      iv: Array.from(iv)
    };
  }

  async decrypt(encryptedData) {
    const encoder = new TextEncoder();
    const decoder = new TextDecoder();
    
    const key = await webcrypto.subtle.importKey(
      'raw',
      encoder.encode(this.secretKey.slice(0, 32)),
      { name: 'AES-GCM' },
      false,
      ['decrypt']
    );
    
    const decrypted = await webcrypto.subtle.decrypt(
      { name: 'AES-GCM', iv: new Uint8Array(encryptedData.iv) },
      key,
      new Uint8Array(encryptedData.data)
    );
    
    return JSON.parse(decoder.decode(decrypted));
  }
}
```

```javascript
// src/utils/cache.js
export class CacheManager {
  constructor(kv) {
    this.kv = kv;
  }

  async get(key, defaultValue = null) {
    try {
      const cached = await this.kv.get(key, 'json');
      return cached !== null ? cached : defaultValue;
    } catch (error) {
      console.error('Cache get error:', error);
      return defaultValue;
    }
  }

  async set(key, value, ttl = 3600) {
    try {
      await this.kv.put(key, JSON.stringify(value), {
        expirationTtl: ttl
      });
    } catch (error) {
      console.error('Cache set error:', error);
    }
  }

  async delete(key) {
    try {
      await this.kv.delete(key);
    } catch (error) {
      console.error('Cache delete error:', error);
    }
  }

  async getOrSet(key, factory, ttl = 3600) {
    let value = await this.get(key);
    if (value === null) {
      value = await factory();
      await this.set(key, value, ttl);
    }
    return value;
  }
}
```

```javascript
// src/utils/database.js
export class DatabaseManager {
  constructor(db) {
    this.db = db;
  }

  async batchInsert(table, records, batchSize = 100) {
    const batches = [];
    for (let i = 0; i < records.length; i += batchSize) {
      batches.push(records.slice(i, i + batchSize));
    }

    for (const batch of batches) {
      const placeholders = batch.map(() => '(?)').join(',');
      const query = `INSERT INTO ${table} VALUES ${placeholders}`;
      
      await this.db.prepare(query)
        .bind(...batch.map(record => JSON.stringify(record)))
        .run();
    }
  }

  async executeWithRetry(query, params = [], maxRetries = 3) {
    for (let attempt = 1; attempt <= maxRetries; attempt++) {
      try {
        return await this.db.prepare(query).bind(...params).run();
      } catch (error) {
        if (attempt === maxRetries) throw error;
        
        // 指数退避
        const delay = Math.pow(2, attempt) * 100;
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }
  }
}
```
