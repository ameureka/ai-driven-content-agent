# AI驱动内容代理系统 - 代码规范

## 📋 概述

本文档定义了AI驱动内容代理系统项目的代码规范，旨在确保代码的一致性、可读性和可维护性。所有团队成员都应严格遵循这些规范。

## 🎯 规范原则

### 核心原则
1. **一致性**: 整个项目保持统一的代码风格
2. **可读性**: 代码应该易于理解和维护
3. **简洁性**: 避免不必要的复杂性
4. **安全性**: 遵循安全编码最佳实践
5. **性能**: 考虑代码的执行效率

### 注释规范

#### JSDoc 注释
```javascript
/**
 * 生成基于模板的内容
 * 
 * @description 使用指定的模板和数据生成格式化的内容
 * @param {Template} template - 内容模板对象
 * @param {ContentData} data - 用于填充模板的数据
 * @param {GenerationOptions} [options] - 可选的生成选项
 * @returns {Promise<GeneratedContent>} 生成的内容对象
 * 
 * @throws {TemplateError} 当模板格式无效时抛出
 * @throws {ValidationError} 当数据验证失败时抛出
 * 
 * @example
 * ```javascript
 * const template = await loadTemplate('article');
 * const data = { title: '标题', content: '内容' };
 * const result = await generateContent(template, data);
 * console.log(result.html);
 * ```
 * 
 * @since 1.0.0
 * @author 开发团队
 */
async function generateContent(template, data, options = {}) {
  // 验证输入参数
  validateTemplate(template);
  validateContentData(data);
  
  try {
    // 预处理数据
    const processedData = preprocessData(data, options);
    
    // 渲染模板
    const renderedContent = await renderTemplate(template, processedData);
    
    // 后处理内容
    const finalContent = postprocessContent(renderedContent, options);
    
    return {
      html: finalContent.html,
      metadata: finalContent.metadata,
      generatedAt: new Date().toISOString()
    };
  } catch (error) {
    logger.error('Content generation failed', { template: template.id, error });
    throw new ContentGenerationError('Failed to generate content', error);
  }
}
```

    repository: Repository<Content>,
    cache: CacheService
  ) {
    this.repository = repository;
    this.cache = cache;
  }
  
  public async getById(id: string): Promise<Content | null> {
    // 先检查缓存
    const cached = await this.cache.get<Content>(`content:${id}`);
    if (cached) {
      return cached;
    }
    
    // 从数据库获取
    const content = await this.repository.findById(id);
    if (content) {
      await this.cache.set(`content:${id}`, content, 3600); // 缓存1小时
    }
    
    return content;
  }

// ✅ Result 模式
type Result<T, E = Error> = 
  | { success: true; data: T }
  | { success: false; error: E };

function safeParseJson<T>(json: string): Result<T> {
  try {
    const data = JSON.parse(json) as T;
    return { success: true, data };
  } catch (error) {
    return { success: false, error: error as Error };
  }
}

// api.integration.test.js
import { describe, it, expect, beforeAll, afterAll, beforeEach } from 'vitest';
import request from 'supertest';
import { app } from '../app.js';
import { setupTestDatabase, cleanupTestDatabase } from './helpers/database.js';
import { createTestUser, createTestContent } from './helpers/factories.js';

describe('Content API Integration Tests', () => {
  let testUser;
  let authToken;
  
  beforeAll(async () => {
    await setupTestDatabase();
  });
  
  afterAll(async () => {
    await cleanupTestDatabase();
  });
  
  beforeEach(async () => {
    testUser = await createTestUser();
    authToken = generateAuthToken(testUser);
  });
  
  describe('POST /api/v1/content', () => {
    it('should create new content with valid data', async () => {
      // Arrange
      const contentData = {
        title: 'Integration Test Content',
        body: 'This is integration test content',
        templateId: 'template-123'
      };
      
      // Act
      const response = await request(app)
        .post('/api/v1/content')
        .set('Authorization', `Bearer ${authToken}`)
        .send(contentData)
        .expect(201);
      
      // Assert
      expect(response.body).toMatchObject({
        success: true,
        data: {
          id: expect.any(String),
          title: contentData.title,
          body: contentData.body,
          authorId: testUser.id,
          status: 'draft',
          createdAt: expect.any(String),
          updatedAt: expect.any(String)
        }
      });
    });
    
    it('should return 400 for invalid content data', async () => {
      // Arrange
      const invalidData = {
        title: '', // 空标题
        body: 'Content body'
      };
      
      // Act
      const response = await request(app)
        .post('/api/v1/content')
        .set('Authorization', `Bearer ${authToken}`)
        .send(invalidData)
        .expect(400);
      
      // Assert
      expect(response.body).toMatchObject({
        success: false,
        message: 'Validation failed',
        errors: expect.arrayContaining([
          expect.objectContaining({
            field: 'title',
            message: expect.any(String)
          })
        ])
      });
    });
    
    it('should return 401 for unauthenticated requests', async () => {
      // Arrange
      const contentData = {
        title: 'Test Content',
        body: 'Content body'
      };
      
      // Act & Assert
      await request(app)
        .post('/api/v1/content')
        .send(contentData)
        .expect(401);
    });
  });
  
  describe('GET /api/v1/content/:id', () => {
    it('should return content by id', async () => {
      // Arrange
      const content = await createTestContent({ authorId: testUser.id });
      
      // Act
      const response = await request(app)
        .get(`/api/v1/content/${content.id}`)
        .set('Authorization', `Bearer ${authToken}`)
        .expect(200);
      
      // Assert
      expect(response.body).toMatchObject({
        success: true,
        data: {
          id: content.id,
          title: content.title,
          body: content.body,
          authorId: testUser.id
        }
      });
    });
    
    it('should return 404 for non-existent content', async () => {

  });
