# 通用渲染模板设计原则与最佳实践

## 概述

本文档总结了设计和开发通用渲染模板的核心原则和最佳实践，适用于各种平台和内容类型的模板开发。这些原则基于实际项目经验，具有广泛的通用性和可操作性。

## 1. 设计原则基础

### 1.1 平台约束理解

#### 微信公众号平台
- **CSS限制**: 仅支持内联样式，禁用外部CSS和`<style>`标签
- **JavaScript限制**: 完全禁用，所有交互通过纯HTML/CSS实现
- **HTML结构**: 需兼容编辑器解析规则，避免复杂嵌套
- **标签限制**: 部分HTML5标签不支持，需使用基础标签
- **属性限制**: 某些HTML属性可能被过滤或修改
- **编辑器兼容**: 必须在微信编辑器中正常显示和编辑

#### 通用Web平台
- **响应式设计**: 支持多设备适配
- **语义化HTML**: 提供良好的SEO和可访问性
- **性能优化**: 考虑加载速度和渲染效率

#### 移动端原生应用
- **WebView兼容**: 考虑不同WebView内核差异
- **触摸交互**: 优化触摸目标大小和间距
- **性能约束**: 内存和CPU使用限制

## 3. 内容处理核心算法

### 3.1 智能标题处理

```javascript
class TitleProcessor {
  constructor() {
    this.counters = [0, 0, 0, 0, 0, 0]; // 支持6级标题
    this.skipFirstTitle = false;
  }
  
  processTitle(content, options = {}) {
    const { skipFirst = false, format = 'number' } = options;
    this.skipFirstTitle = skipFirst;
    
    return content.replace(/^(#{1,6})\s+(.+)$/gm, (match, hashes, title) => {
      const level = hashes.length;
      
      // 跳过第一个标题逻辑
      if (this.skipFirstTitle) {
        this.skipFirstTitle = false;
        return match; // 保持原样或移除
      }
      
      // 更新计数器
      this.updateCounters(level);
      
      // 生成编号
      const number = this.generateNumber(level, format);
      const id = this.generateId(title);
      
      return this.formatTitle(hashes, number, title, id);
    });
  }
  
  updateCounters(level) {
    this.counters[level - 1]++;
    for (let i = level; i < this.counters.length; i++) {
      this.counters[i] = 0;
    }
  }
  
  generateNumber(level, format) {
    const numbers = this.counters.slice(0, level).filter(c => c > 0);
    
    switch (format) {
      case 'number':
        return numbers.join('.');
      case 'chinese':
        return this.toChineseNumber(numbers);
      case 'roman':
        return this.toRomanNumber(numbers[0]);
      default:
        return numbers.join('.');
    }
  }
}
```

### 3.2 目录生成算法

```javascript
class TOCGenerator {
  generateTOC(content, options = {}) {
    const {
      maxLevel = 3,
      skipFirst = false,
      style = 'nested',
      enableLinks = true
    } = options;
    
    const headings = this.extractHeadings(content, maxLevel, skipFirst);
    
    switch (style) {
      case 'flat':
        return this.generateFlatTOC(headings, enableLinks);
      case 'nested':
        return this.generateNestedTOC(headings, enableLinks);
      case 'accordion':
        return this.generateAccordionTOC(headings, enableLinks);
      default:
        return this.generateNestedTOC(headings, enableLinks);
    }
  }
  
  extractHeadings(content, maxLevel, skipFirst) {
    const headings = [];
    const lines = content.split('\n');
    let firstSkipped = !skipFirst;
    
    lines.forEach((line, index) => {
      const match = line.match(/^(#{1,6})\s+(.+)$/);
      if (match) {
        const level = match[1].length;
        if (level <= maxLevel) {
          if (!firstSkipped) {
            firstSkipped = true;
            return;
          }
          
          headings.push({
            level,
            title: match[2],
            id: `heading-${index}`,
            line: index
          });
        }
      }
    });
    
    return headings;
  }
}
```

```javascript
  optimizeTemplate(template) {
    // CSS优化
    template = this.minifyStyles(template);

  minifyStyles(template) {
    return template.replace(/\s*:\s*/g, ':')
                  .replace(/;\s*/g, ';')
                  .replace(/\s*{\s*/g, '{')

    return template.replace(
      /<img([^>]*?)src="([^"]*)"([^>]*?)>/g,
      '<img$1loading="lazy" src="$2"$3>'
    );
  }
}
```

```javascript
class MemoryManager {
  constructor() {
    this.cache = new Map();
    this.maxCacheSize = 100;
  }
  
  memoize(fn, key) {
    if (this.cache.has(key)) {
      return this.cache.get(key);
    }
    
    const result = fn();
    
    if (this.cache.size >= this.maxCacheSize) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
    
    this.cache.set(key, result);
    return result;
  }
}
```

```javascript
class TemplateValidator {
  async validateTemplate(template, testCases) {
    const results = [];
    
    for (const testCase of testCases) {
      const result = await this.runTest(template, testCase);
      results.push(result);
    }
    
    return this.generateReport(results);
  }
  
  async runTest(template, testCase) {
    const { input, expected, description } = testCase;
    
    try {
      const output = template.render(input);
      const isValid = this.compareOutput(output, expected);
      
      return {
        description,
        passed: isValid,
        input,
        output,
        expected
      };
    } catch (error) {
      return {
        description,
        passed: false,
        error: error.message
      };
    }
  }
}
```

```javascript
class PerformanceBenchmark {
  async benchmark(template, iterations = 100) {
    const results = [];
    
    for (let i = 0; i < iterations; i++) {
      const start = performance.now();
      template.render(this.getTestData());
      const end = performance.now();
      
      results.push(end - start);
    }
    
    return {
      average: results.reduce((a, b) => a + b) / results.length,
      min: Math.min(...results),
      max: Math.max(...results),
      iterations
    };
  }
}
```

```javascript
class I18nManager {
  constructor() {
    this.translations = new Map();
  }
  
  addTranslation(locale, key, value) {
    if (!this.translations.has(locale)) {
      this.translations.set(locale, new Map());
    }
    this.translations.get(locale).set(key, value);
  }
  
  translate(key, locale = 'zh-CN') {
    const translations = this.translations.get(locale);
    return translations ? translations.get(key) || key : key;

}
```
