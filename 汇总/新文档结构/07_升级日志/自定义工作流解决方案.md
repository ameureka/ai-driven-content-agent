## 🔧 技术实现细节

### 后端核心代码

```javascript
class WorkflowManager {
  constructor() {
    this.defaultWorkflows = {
      'dify-general': {
        name: 'URL内容生成',
        type: 'url',
        envKey: 'DIFY_API_KEY'
      },
      'dify-article': {
        name: 'AI文章生成', 
        type: 'text',
        envKey: 'DIFY_ARTICLE_API_KEY'
      }
    };
    this.customWorkflows = this.loadCustomWorkflows();
  }

  getWorkflowConfig(workflowId) {
    return this.defaultWorkflows[workflowId] || 
           this.customWorkflows[workflowId] ||
           null;
  }

  async executeWorkflow(workflowId, inputs, options) {
    const workflow = this.getWorkflowConfig(workflowId);
    const apiKey = workflow.isCustom ? workflow.apiKey : env[workflow.envKey];
    
    if (workflow.type === 'url') {
      return options.stream 
        ? callDifyWorkflowStreaming(inputs.url, apiKey, options.callbacks)
        : callDifyWorkflow(inputs.url, apiKey);
    } else {
      return options.stream
        ? callDifyArticleWorkflow(inputs, apiKey, options.callbacks)  
        : callDifyArticleWorkflow(inputs, apiKey);
    }
  }
}
```

### 前端核心代码

```javascript
class CustomWorkflowUI {
  constructor() {
    this.currentWorkflow = 'dify-general';
    this.availableWorkflows = new Map();
    this.init();
  }

  async loadWorkflows() {
    const response = await fetch('/api/v1/workflows/available');
    const { data } = await response.json();
    data.forEach(workflow => {
      this.availableWorkflows.set(workflow.id, workflow);
    });
    this.renderWorkflowSelector();
  }

  selectWorkflow(workflowId) {
    this.currentWorkflow = workflowId;
    this.updateInputFields();
    this.updateUI();
  }

  async executeWorkflow(inputs) {
    const apiUrl = `/api/v1/workflows/${this.currentWorkflow}/execute?stream=true`;
    // 使用现有的流式处理逻辑
    return await fetchWithRetryUrl(apiUrl, { inputs }, apiKey);
  }
}
```
