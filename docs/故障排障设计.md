```toml
id = "your-namespace-id"
preview_id = "your-preview-namespace-id"
```


### 安全相关问题

#### 问题8: CORS错误

**症状:**
```
Access to fetch at 'https://your-domain.workers.dev' from origin 'https://example.com' has been blocked by CORS policy
```

**解决方案:**

1. **配置CORS头**
```javascript
// 在index.js中添加CORS处理
function addCORSHeaders(response) {
  response.headers.set('Access-Control-Allow-Origin', '*');
  response.headers.set('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
  response.headers.set('Access-Control-Allow-Headers', 'Content-Type, Authorization');
  response.headers.set('Access-Control-Max-Age', '86400');
  return response;
}

// 处理OPTIONS请求
if (request.method === 'OPTIONS') {
  return addCORSHeaders(new Response(null, { status: 204 }));
}
```

2. **限制允许的域名**
```javascript
// 更安全的CORS配置
const allowedOrigins = [
  'https://your-domain.com',
  'https://www.your-domain.com',
  'http://localhost:3000' // 开发环境
];

function getCORSHeaders(request) {
  const origin = request.headers.get('Origin');
  const isAllowed = allowedOrigins.includes(origin);
  
  return {
    'Access-Control-Allow-Origin': isAllowed ? origin : 'null',
    'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
    'Access-Control-Allow-Headers': 'Content-Type',
    'Vary': 'Origin'
  };
}
```

```javascript
// 大对象分片存储
async function putLargeObject(key, data) {
  const serialized = JSON.stringify(data);
  const chunkSize = 1024 * 1024; // 1MB chunks
  const chunks = [];
  
  for (let i = 0; i < serialized.length; i += chunkSize) {
    chunks.push(serialized.slice(i, i + chunkSize));
  }
  
  // 存储元数据
  await CONTENT_CACHE.put(`${key}_meta`, JSON.stringify({
    chunks: chunks.length,
    totalSize: serialized.length
  }));
  
  // 存储分片
  for (let i = 0; i < chunks.length; i++) {
    await CONTENT_CACHE.put(`${key}_chunk_${i}`, chunks[i]);
  }
}
```

```javascript
// 增强的健康检查
async function healthCheck(env) {
  const checks = {
    timestamp: new Date().toISOString(),
    status: 'ok',

      headers: { 'Authorization': `Bearer ${env.DIFY_API_KEY}` }
    });
    checks.checks.dify = response.ok ? 'healthy' : 'unhealthy';

    checks.checks.dify = 'unhealthy';
  }
  
  // 模板检查
  try {
    const templates = await TemplateManager.getAvailableTemplates();
    checks.checks.templates = templates.length > 0 ? 'healthy' : 'unhealthy';
  } catch (error) {
    checks.checks.templates = 'unhealthy';
  }
  
  // 整体状态
  const allHealthy = Object.values(checks.checks).every(status => status === 'healthy');
  checks.status = allHealthy ? 'ok' : 'degraded';
```

### 2. 错误追踪

```javascript
// 错误追踪和报告
class ErrorTracker {
  static async logError(error, context = {}) {
    const errorData = {
      timestamp: new Date().toISOString(),
      message: error.message,
      stack: error.stack,
      context,
      userAgent: context.request?.headers?.get('User-Agent'),
      url: context.request?.url
    };
    
    // 记录到KV存储
    const errorKey = `error_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
    await CONTENT_CACHE.put(errorKey, JSON.stringify(errorData), {
      expirationTtl: 7 * 24 * 3600 // 保留7天
    });
    
    // 控制台输出
    console.error('Error logged:', errorData);
    
    // 可选：发送到外部监控服务
    // await this.sendToMonitoring(errorData);
  }
  
  static async getRecentErrors(limit = 10) {
    // 获取最近的错误记录
    const keys = await CONTENT_CACHE.list({ prefix: 'error_' });
    const errors = [];
    
    for (const key of keys.keys.slice(0, limit)) {
      const errorData = await CONTENT_CACHE.get(key.name);
      if (errorData) {
        errors.push(JSON.parse(errorData));
      }
    }
    
    return errors.sort((a, b) => new Date(b.timestamp) - new Date(a.timestamp));
  }
}
```

### 3. 性能监控

```javascript
// 性能指标收集
class PerformanceMonitor {
  static async recordMetric(name, value, tags = {}) {
    const metric = {
      name,
      value,
      tags,
      timestamp: new Date().toISOString()
    };
    
    // 存储到KV（可选择性存储）
    if (Math.random() < 0.1) { // 10%采样率
      const metricKey = `metric_${name}_${Date.now()}`;
      await CONTENT_CACHE.put(metricKey, JSON.stringify(metric), {
        expirationTtl: 24 * 3600 // 保留24小时
      });
    }
    
    console.log('Metric recorded:', metric);
  }
  
  static async getMetrics(name, hours = 1) {
    const keys = await CONTENT_CACHE.list({ prefix: `metric_${name}_` });
    const metrics = [];
    
    const cutoff = new Date(Date.now() - hours * 3600 * 1000);
    
    for (const key of keys.keys) {
      const metricData = await CONTENT_CACHE.get(key.name);
      if (metricData) {
        const metric = JSON.parse(metricData);
        if (new Date(metric.timestamp) > cutoff) {
          metrics.push(metric);
        }
      }
    }
    
    return metrics;
  }
}
```

```bash
response_time=$(curl -s -o /dev/null -w "%{time_total}" "$API_ENDPOINT/api/status")
if (( $(echo "$response_time > 2.0" | bc -l) )); then
  curl -X POST -H 'Content-type: application/json' \
    --data "{\"text\":\"⚠️ AI Content Agent response time: ${response_time}s\"}" \
    "$SLACK_WEBHOOK"
```

### 3. 容量规划

```javascript
// 使用量统计
class UsageTracker {
  static async recordUsage(type, amount = 1) {
    const today = new Date().toISOString().split('T')[0];
    const key = `usage_${type}_${today}`;
    
    const current = await CONTENT_CACHE.get(key) || '0';
    const newValue = parseInt(current) + amount;
    
    await CONTENT_CACHE.put(key, newValue.toString(), {
      expirationTtl: 7 * 24 * 3600 // 保留7天
    });
  }
  
  static async getUsageReport(days = 7) {
    const report = {};
    const types = ['api_calls', 'template_renders', 'cache_hits'];
    
    for (let i = 0; i < days; i++) {
      const date = new Date(Date.now() - i * 24 * 3600 * 1000)
        .toISOString().split('T')[0];
      
      report[date] = {};
      
      for (const type of types) {
        const key = `usage_${type}_${date}`;
        const value = await CONTENT_CACHE.get(key) || '0';
        report[date][type] = parseInt(value);
      }
    }
    
    return report;
  }
}
```
