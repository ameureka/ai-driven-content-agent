# AI驱动内容代理系统 - 测试策略与规范

## 📋 概述

本文档定义了AI驱动内容代理系统的完整测试策略，包括测试金字塔模型、质量指标、测试工具链、最佳实践和质量门禁标准。

## 🏗️ 测试架构设计

### 测试金字塔模型

我们采用经典的测试金字塔模型，确保测试的效率和覆盖面：

```
        🔺 E2E测试 (5%)
       /              \
      /   集成测试 (25%)   \
     /                    \
    /    单元测试 (70%)      \
   /________________________\
```

#### 测试层级分布

| 测试类型 | 占比 | 执行速度 | 维护成本 | 反馈速度 | 主要目的 |
|----------|------|----------|----------|----------|----------|
| **单元测试** | 70% | 极快 | 低 | 秒级 | 验证单个函数/组件逻辑 |
| **集成测试** | 25% | 中等 | 中等 | 分钟级 | 验证模块间交互 |
| **端到端测试** | 5% | 较慢 | 高 | 分钟级 | 验证完整用户流程 |

```javascript
import request from 'supertest';
import app from '../app';
import { setupTestDB, cleanupTestDB } from '../test/helpers/database';

describe('Content Rendering Integration', () => {
  beforeAll(async () => {
    await setupTestDB();
  });

  afterAll(async () => {
    await cleanupTestDB();
  });

  test('完整的内容渲染流程', async () => {
    // 1. 上传测试内容

      })

    const contentId = uploadResponse.body.contentId;

    // 2. 渲染内容
    const renderResponse = await request(app)
      .post('/api/v1/render')
      .send({
        contentId,
        template: 'article_wechat',
        apiKey: process.env.TEST_API_KEY
      })
      .expect(200);

    expect(renderResponse.body.success).toBe(true);
    expect(renderResponse.body.renderedContent).toContain('测试文章');

    // 3. 验证渲染结果存储
    const resultResponse = await request(app)
      .get(`/api/v1/render/${renderResponse.body.renderId}`)
      .expect(200);

    expect(resultResponse.body.status).toBe('completed');
  });
```

## 🎯 测试最佳实践

### 测试编写原则

#### 1. AAA模式 (Arrange-Act-Assert)
```javascript
test('应该正确计算文章字数', () => {
  // Arrange - 准备测试数据
  const article = {
    title: '测试文章',
    content: '这是一篇包含中文和English的测试文章。'
  };
  
  // Act - 执行被测试的操作
  const wordCount = calculateWordCount(article);
  
  // Assert - 验证结果
  expect(wordCount).toBe(12);
});
```

#### 2. 测试命名规范
```javascript
// ✅ 好的测试命名
describe('ContentRenderer', () => {
  describe('render方法', () => {
    test('当提供有效模板时应该返回渲染结果', () => {});
    test('当模板不存在时应该抛出错误', () => {});
    test('当内容为空时应该返回空字符串', () => {});
  });
});

// ❌ 不好的测试命名
describe('ContentRenderer', () => {
  test('test1', () => {});
  test('render测试', () => {});
  test('应该工作', () => {});
});
```

```javascript
// test/reporters/dashboard-reporter.js
class DashboardReporter {
  constructor(globalConfig, options) {
    this.globalConfig = globalConfig;
    this.options = options;
  }
  
  onRunComplete(contexts, results) {
    const summary = {
      timestamp: new Date().toISOString(),
      totalTests: results.numTotalTests,
      passedTests: results.numPassedTests,
      failedTests: results.numFailedTests,
      coverage: this.extractCoverage(results),
      duration: results.testResults.reduce((acc, result) => 
        acc + result.perfStats.end - result.perfStats.start, 0
      ),
      environment: process.env.NODE_ENV || 'test'
    };
    
    // 发送到监控系统
    this.sendToMonitoring(summary);
    
    // 生成HTML报告
    this.generateHTMLReport(summary, results);
  }
  
  extractCoverage(results) {
    const coverage = results.coverageMap;
    if (!coverage) return null;
    
    const summary = coverage.getCoverageSummary();
    return {
      lines: summary.lines.pct,
      functions: summary.functions.pct,
      branches: summary.branches.pct,
      statements: summary.statements.pct
    };
  }
  
  async sendToMonitoring(summary) {
    try {
      await fetch(process.env.MONITORING_ENDPOINT, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          metric: 'test_results',
          data: summary,
          tags: {
            project: 'ai-content-agent',
            environment: summary.environment
          }
        })
      });
    } catch (error) {
      console.warn('无法发送测试结果到监控系统:', error.message);
    }
  }
}

module.exports = DashboardReporter;
```

```javascript
import { performance } from 'perf_hooks';
import { ContentRenderer } from '../../src/services/ContentRenderer';

describe('性能基准测试', () => {

  test('内容渲染性能应该在可接受范围内', async () => {
    const iterations = 100;
    const times = [];
    
    for (let i = 0; i < iterations; i++) {
      const start = performance.now();
      await renderer.render(testContent, 'article_wechat');
      const end = performance.now();
      times.push(end - start);
    }
    
    const avgTime = times.reduce((a, b) => a + b) / times.length;
    const p95Time = times.sort((a, b) => a - b)[Math.floor(times.length * 0.95)];
    
    console.log(`平均渲染时间: ${avgTime.toFixed(2)}ms`);
    console.log(`P95渲染时间: ${p95Time.toFixed(2)}ms`);
    
    // 性能断言
    expect(avgTime).toBeLessThan(1000); // 平均时间小于1秒
    expect(p95Time).toBeLessThan(2000);  // P95时间小于2秒
  });
  
  test('内存使用应该稳定', async () => {
    const initialMemory = process.memoryUsage().heapUsed;
    
    // 执行多次渲染
    for (let i = 0; i < 50; i++) {
      await renderer.render(testContent, 'article_wechat');
    }
    
    // 强制垃圾回收
    if (global.gc) {
      global.gc();
    }
    
    const finalMemory = process.memoryUsage().heapUsed;

    console.log(`内存增长: ${(memoryIncrease / 1024 / 1024).toFixed(2)}MB`);
```

### 测试质量评估

```javascript
// scripts/test-quality-analysis.js
const fs = require('fs');
const path = require('path');

class TestQualityAnalyzer {
  constructor() {
    this.metrics = {
      testCoverage: 0,
      testToCodeRatio: 0,
      averageTestComplexity: 0,
      duplicatedTestCode: 0,
      testMaintainabilityIndex: 0
    };
  }
  
  analyzeTestSuite() {
    const testFiles = this.findTestFiles();
    const sourceFiles = this.findSourceFiles();
    
    this.calculateTestToCodeRatio(testFiles, sourceFiles);
    this.analyzeTestComplexity(testFiles);
    this.detectDuplicatedCode(testFiles);
    this.calculateMaintainabilityIndex(testFiles);
    
    return this.generateReport();
  }
  
  calculateTestToCodeRatio(testFiles, sourceFiles) {
    const testLOC = this.countLinesOfCode(testFiles);
    const sourceLOC = this.countLinesOfCode(sourceFiles);
    
    this.metrics.testToCodeRatio = testLOC / sourceLOC;
  }
  
  analyzeTestComplexity(testFiles) {
    let totalComplexity = 0;
    let testCount = 0;
    
    testFiles.forEach(file => {
      const content = fs.readFileSync(file, 'utf8');
      const complexity = this.calculateCyclomaticComplexity(content);
      const tests = this.countTests(content);
      
      totalComplexity += complexity;
      testCount += tests;
    });
    
    this.metrics.averageTestComplexity = totalComplexity / testCount;
  }
  
  generateReport() {
    const report = {
      timestamp: new Date().toISOString(),
      metrics: this.metrics,
      recommendations: this.generateRecommendations(),
      score: this.calculateOverallScore()
    };
    
    // 保存报告
    fs.writeFileSync(
      'test-quality-report.json',
      JSON.stringify(report, null, 2)
    );
    
    return report;
  }
  
  generateRecommendations() {
    const recommendations = [];
    
    if (this.metrics.testToCodeRatio < 0.5) {
      recommendations.push('建议增加测试代码覆盖率，当前测试代码比例偏低');
    }
    
    if (this.metrics.averageTestComplexity > 10) {
      recommendations.push('部分测试用例过于复杂，建议拆分为更小的测试');
    }
    
    if (this.metrics.duplicatedTestCode > 20) {
      recommendations.push('存在较多重复的测试代码，建议提取公共测试工具函数');
    }
    
    return recommendations;
  }
}

// 运行分析
const analyzer = new TestQualityAnalyzer();
const report = analyzer.analyzeTestSuite();
console.log('测试质量分析完成，报告已保存到 test-quality-report.json');
```
