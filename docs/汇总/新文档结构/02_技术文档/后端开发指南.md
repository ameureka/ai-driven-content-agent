# 后端开发指南

## 概述

本文档详细介绍了AI驱动内容代理系统后端开发的技术栈、架构设计、开发规范和最佳实践。后端基于Cloudflare Workers平台，使用Hono.js框架构建高性能的边缘计算API服务。

## 技术栈

### 核心技术

- **运行时**: Cloudflare Workers (V8 JavaScript Engine)
- **框架**: Hono.js - 轻量级、高性能的Web框架
- **语言**: TypeScript - 类型安全的JavaScript超集
- **数据库**: Cloudflare D1 - 基于SQLite的边缘数据库
- **缓存**: Cloudflare KV - 全球分布式键值存储
- **文件存储**: Cloudflare R2 - 对象存储服务
- **向量搜索**: Cloudflare Vectorize - AI向量数据库

### 开发工具

- **包管理**: npm/pnpm
- **构建工具**: Wrangler CLI
- **测试框架**: Vitest + Miniflare
- **代码规范**: ESLint + Prettier
- **类型检查**: TypeScript Compiler
- **API文档**: OpenAPI/Swagger

```typescript
// src/middleware/auth.ts
import { createMiddleware } from 'hono/factory';
import { verify } from 'hono/jwt';
import type { Context } from '../types/common';

export const authMiddleware = createMiddleware<Context>(async (c, next) => {
  try {
    const authHeader = c.req.header('Authorization');
    
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      return c.json({ error: 'Unauthorized', message: 'Missing or invalid authorization header' }, 401);
    }
    
    const token = authHeader.substring(7); // Remove 'Bearer ' prefix
    
    // 验证JWT令牌
    const payload = await verify(token, c.env.JWT_SECRET);
    
    if (!payload || !payload.sub) {
      return c.json({ error: 'Unauthorized', message: 'Invalid token' }, 401);
    }
    
    // 检查令牌是否在黑名单中
    const blacklisted = await c.env.CACHE.get(`blacklist:${token}`);
    if (blacklisted) {
      return c.json({ error: 'Unauthorized', message: 'Token has been revoked' }, 401);
    }
    
    // 从数据库获取用户信息
    const user = await c.env.DB.prepare(
      'SELECT id, email, role, status FROM users WHERE id = ? AND status = "active"'
    ).bind(payload.sub).first();
    
    if (!user) {
      return c.json({ error: 'Unauthorized', message: 'User not found or inactive' }, 401);
    }
    
    // 设置用户信息到上下文
    c.set('user', {
      id: user.id as string,
      email: user.email as string,
      role: user.role as string,
    });
    
    await next();
  } catch (error) {
    console.error('Auth middleware error:', error);
    return c.json({ error: 'Unauthorized', message: 'Token verification failed' }, 401);
  }
});
```

### 限流中间件

```typescript
// src/middleware/rateLimit.ts
import { createMiddleware } from 'hono/factory';
import type { Context } from '../types/common';

interface RateLimitOptions {
  windowMs: number;
  maxRequests: number;
  keyGenerator?: (c: any) => string;
}

export const createRateLimitMiddleware = (options: RateLimitOptions) => {
  return createMiddleware<Context>(async (c, next) => {
    try {
      const key = options.keyGenerator ? options.keyGenerator(c) : getClientIP(c);
      const windowKey = `ratelimit:${key}:${Math.floor(Date.now() / options.windowMs)}`;
      
      // 获取当前窗口的请求计数
      const currentCount = await c.env.CACHE.get(windowKey);
      const count = currentCount ? parseInt(currentCount) : 0;
      
      if (count >= options.maxRequests) {
        return c.json({
          error: 'Too Many Requests',
          message: 'Rate limit exceeded. Please try again later.',
          retryAfter: Math.ceil(options.windowMs / 1000),
        }, 429);
      }
      
      // 增加计数
      await c.env.CACHE.put(windowKey, (count + 1).toString(), {
        expirationTtl: Math.ceil(options.windowMs / 1000),
      });
      
      // 设置响应头
      c.header('X-RateLimit-Limit', options.maxRequests.toString());
      c.header('X-RateLimit-Remaining', (options.maxRequests - count - 1).toString());
      c.header('X-RateLimit-Reset', (Date.now() + options.windowMs).toString());
      
      await next();
    } catch (error) {
      console.error('Rate limit middleware error:', error);
      await next(); // 在错误情况下继续处理请求
    }
  });
};

// 默认限流中间件
export const rateLimitMiddleware = createRateLimitMiddleware({
  windowMs: 15 * 60 * 1000, // 15分钟
  maxRequests: 100, // 每个窗口最多100个请求
});

// 获取客户端IP
function getClientIP(c: any): string {
  return c.req.header('CF-Connecting-IP') || 
         c.req.header('X-Forwarded-For') || 
         c.req.header('X-Real-IP') || 
         'unknown';
}
```

```typescript
import { createMiddleware } from 'hono/factory';
import { z } from 'zod';
import type { Context } from '../types/common';

export const validateBody = <T extends z.ZodType>(schema: T) => {
  return createMiddleware<Context>(async (c, next) => {
    try {
      const body = await c.req.json();
      const result = schema.safeParse(body);
      
      if (!result.success) {
        return c.json({
          error: 'Validation Error',
          message: 'Request body validation failed',
          details: result.error.issues.map(issue => ({
            path: issue.path.join('.'),
            message: issue.message,
            code: issue.code,
          })),
        }, 400);
      }
      
      // 将验证后的数据存储到上下文中
      c.set('validatedBody', result.data);
      await next();
    } catch (error) {
      return c.json({
        error: 'Invalid JSON',
        message: 'Request body must be valid JSON',
      }, 400);
    }
  });
};

export const validateQuery = <T extends z.ZodType>(schema: T) => {
  return createMiddleware<Context>(async (c, next) => {
    try {
      const query = c.req.query();
      const result = schema.safeParse(query);
      
      if (!result.success) {
        return c.json({
          error: 'Validation Error',
          message: 'Query parameters validation failed',
          details: result.error.issues.map(issue => ({
            path: issue.path.join('.'),
            message: issue.message,
            code: issue.code,
          })),
        }, 400);
      }
      
      c.set('validatedQuery', result.data);
      await next();
    } catch (error) {
      return c.json({
        error: 'Validation Error',
        message: 'Query parameter validation failed',
      }, 400);
    }
  });
```

## 业务服务层

### 认证服务

```typescript
// src/services/authService.ts
import { sign, verify } from 'hono/jwt';
import { hash, compare } from 'bcryptjs';
import { nanoid } from 'nanoid';
import type { Bindings } from '../types/common';
import { DatabaseService } from '../utils/database';
import { CacheService } from '../utils/cache';

export interface RegisterData {
  email: string;
  password: string;
  name: string;
}

export interface LoginData {
  email: string;
  password: string;
}

export class AuthService {
  private db: DatabaseService;
  private cache: CacheService;
  private jwtSecret: string;

  constructor(private env: Bindings) {
    this.db = new DatabaseService(env.DB);
    this.cache = new CacheService(env.CACHE);
    this.jwtSecret = env.JWT_SECRET;
  }

  async register(data: RegisterData) {
    // 检查用户是否已存在
    const existingUser = await this.db.findUserByEmail(data.email);
    if (existingUser) {
      throw new Error('User already exists with this email');
    }

    // 密码哈希
    const hashedPassword = await hash(data.password, 12);

    // 创建用户
    const userId = nanoid();
    const user = await this.db.createUser({
      id: userId,
      email: data.email,
      name: data.name,
      password: hashedPassword,
      role: 'user',
      status: 'active',
    });

    // 生成JWT令牌
    const token = await this.generateAccessToken(userId);
    const refreshToken = await this.generateRefreshToken(userId);

    // 存储刷新令牌
    await this.cache.set(`refresh:${userId}`, refreshToken, 7 * 24 * 60 * 60); // 7天

    return {
      user: {
        id: user.id,
        email: user.email,
        name: user.name,
        role: user.role,
      },
      token,
      refreshToken,
    };
  }

  async login(data: LoginData) {
    // 查找用户
    const user = await this.db.findUserByEmail(data.email);
    if (!user || user.status !== 'active') {
      throw new Error('Invalid credentials');
    }

    // 验证密码
    const isValidPassword = await compare(data.password, user.password);
    if (!isValidPassword) {
      throw new Error('Invalid credentials');
    }

    // 更新最后登录时间
    await this.db.updateUserLastLogin(user.id);

    // 生成令牌
    const token = await this.generateAccessToken(user.id);
    const refreshToken = await this.generateRefreshToken(user.id);

    // 存储刷新令牌
    await this.cache.set(`refresh:${user.id}`, refreshToken, 7 * 24 * 60 * 60);

    return {
      user: {
        id: user.id,
        email: user.email,
        name: user.name,
        role: user.role,
      },
      token,
      refreshToken,
    };
  }

  async refreshToken(refreshToken: string) {
    try {
      // 验证刷新令牌
      const payload = await verify(refreshToken, this.jwtSecret);
      if (!payload || !payload.sub) {
        throw new Error('Invalid refresh token');
      }

      const userId = payload.sub as string;

      // 检查存储的刷新令牌
      const storedToken = await this.cache.get(`refresh:${userId}`);
      if (storedToken !== refreshToken) {
        throw new Error('Invalid refresh token');
      }

      // 验证用户状态
      const user = await this.db.findUserById(userId);
      if (!user || user.status !== 'active') {
        throw new Error('User not found or inactive');
      }

      // 生成新令牌
      const newToken = await this.generateAccessToken(userId);
      const newRefreshToken = await this.generateRefreshToken(userId);

      // 更新存储的刷新令牌
      await this.cache.set(`refresh:${userId}`, newRefreshToken, 7 * 24 * 60 * 60);

      return {
        token: newToken,
        refreshToken: newRefreshToken,
      };
    } catch (error) {
      throw new Error('Token refresh failed');
    }
  }

  async logout(token: string) {
    try {
      // 将令牌加入黑名单
      const payload = await verify(token, this.jwtSecret);
      if (payload && payload.exp) {
        const ttl = payload.exp - Math.floor(Date.now() / 1000);
        if (ttl > 0) {
          await this.cache.set(`blacklist:${token}`, 'true', ttl);
        }
      }

      // 删除刷新令牌
      if (payload && payload.sub) {
        await this.cache.delete(`refresh:${payload.sub}`);
      }
    } catch (error) {
      // 忽略错误，继续处理登出
      console.error('Logout error:', error);
    }
  }

  private async generateAccessToken(userId: string): Promise<string> {
    const payload = {
      sub: userId,
      iat: Math.floor(Date.now() / 1000),
      exp: Math.floor(Date.now() / 1000) + (60 * 60), // 1小时
    };

    return await sign(payload, this.jwtSecret);
  }

  private async generateRefreshToken(userId: string): Promise<string> {
    const payload = {
      sub: userId,
      type: 'refresh',
      iat: Math.floor(Date.now() / 1000),
      exp: Math.floor(Date.now() / 1000) + (7 * 24 * 60 * 60), // 7天
    };

    return await sign(payload, this.jwtSecret);
  }
}
```

```typescript
  private db: DatabaseService;
  private dify: DifyService;
  private cache: CacheService;

  constructor(private env: Bindings) {
    this.db = new DatabaseService(env.DB);
    this.dify = new DifyService(env.DIFY_API_KEY, env.DIFY_BASE_URL);
    this.cache = new CacheService(env.CACHE);
  }

  async rewriteContent(request: ContentRewriteRequest) {
    try {
      // 生成缓存键
      const cacheKey = `rewrite:${this.generateCacheKey(request)}`;
      
      // 检查缓存
      const cached = await this.cache.get(cacheKey);
      if (cached) {
        return JSON.parse(cached);
      }

      // 调用Dify API进行内容改写
      const rewrittenContent = await this.dify.rewriteContent({
        content: request.content,
        style: request.style || 'professional',
        tone: request.tone || 'neutral',
        length: request.length || 'medium',
      });

      // 保存到数据库
      const contentId = nanoid();
      await this.db.createContent({
        id: contentId,
        userId: request.userId,
        type: 'rewrite',
        originalContent: request.content,
        processedContent: rewrittenContent,
        parameters: {
          style: request.style,
          tone: request.tone,
          length: request.length,
        },
        status: 'completed',
      });

      const result = {
        id: contentId,
        originalContent: request.content,
        rewrittenContent,
        parameters: {
          style: request.style,
          tone: request.tone,
          length: request.length,
        },
        createdAt: new Date().toISOString(),
      };

      // 缓存结果
      await this.cache.set(cacheKey, JSON.stringify(result), 60 * 60); // 1小时

      return result;
    } catch (error) {
      console.error('Content rewrite error:', error);
      throw new Error('Failed to rewrite content');
    }
  }

  async generateArticle(request: ArticleGenerateRequest) {
    try {
      // 生成缓存键
      const cacheKey = `generate:${this.generateCacheKey(request)}`;
      
      // 检查缓存
      const cached = await this.cache.get(cacheKey);
      if (cached) {
        return JSON.parse(cached);
      }

      // 调用Dify API生成文章
      const article = await this.dify.generateArticle({
        topic: request.topic,
        keywords: request.keywords || [],
        length: request.length || 'medium',
        style: request.style || 'informative',
        outline: request.outline,
      });

      // 保存到数据库
      const contentId = nanoid();
      await this.db.createContent({
        id: contentId,
        userId: request.userId,
        type: 'generate',
        originalContent: request.topic,
        processedContent: article.content,
        parameters: {
          topic: request.topic,
          keywords: request.keywords,
          length: request.length,
          style: request.style,
          outline: request.outline,
        },
        metadata: {
          title: article.title,
          wordCount: article.wordCount,
          readingTime: article.readingTime,
        },
        status: 'completed',
      });

      const result = {
        id: contentId,
        title: article.title,
        content: article.content,
        wordCount: article.wordCount,
        readingTime: article.readingTime,
        parameters: {
          topic: request.topic,
          keywords: request.keywords,
          length: request.length,
          style: request.style,
        },
        createdAt: new Date().toISOString(),
      };

      // 缓存结果

      return result;
    } catch (error) {
      console.error('Article generation error:', error);
      throw new Error('Failed to generate article');
    }
  }

  async analyzeContent({ content, userId }: { content: string; userId?: string }) {
    try {
      // 调用Dify API分析内容
      const analysis = await this.dify.analyzeContent({ content });

      // 保存分析结果
      if (userId) {
        const analysisId = nanoid();
        await this.db.createContent({
          id: analysisId,
          userId,
          type: 'analysis',
          originalContent: content,
          processedContent: JSON.stringify(analysis),
          status: 'completed',
        });
      }

      return analysis;
    } catch (error) {
      console.error('Content analysis error:', error);
      throw new Error('Failed to analyze content');
    }
  }

  async getContentHistory({ userId, page = 1, limit = 20 }: {
    userId?: string;
    page?: number;
    limit?: number;
  }) {
    try {
      if (!userId) {
        throw new Error('User ID is required');
      }

      const offset = (page - 1) * limit;
      const contents = await this.db.getUserContents(userId, limit, offset);
      const total = await this.db.getUserContentCount(userId);

      return {
        contents,
        pagination: {
          page,
          limit,
          total,
          pages: Math.ceil(total / limit),
        },
      };
    } catch (error) {
      console.error('Get content history error:', error);
      throw new Error('Failed to fetch content history');
    }
  }

  private generateCacheKey(data: any): string {
    const str = JSON.stringify(data, Object.keys(data).sort());
    return btoa(str).replace(/[+/=]/g, '').substring(0, 32);
  }
```

```typescript
// src/utils/cache.ts
import type { KVNamespace } from '@cloudflare/workers-types';

export class CacheService {
  constructor(private kv: KVNamespace) {}

  async get(key: string): Promise<string | null> {
    try {
      return await this.kv.get(key);
    } catch (error) {
      console.error('Cache get error:', error);
      return null;
    }
  }

  async set(key: string, value: string, ttl?: number): Promise<void> {
    try {
      const options: any = {};
      if (ttl) {
        options.expirationTtl = ttl;
      }
      await this.kv.put(key, value, options);
    } catch (error) {
      console.error('Cache set error:', error);
    }
  }

  async delete(key: string): Promise<void> {
    try {
      await this.kv.delete(key);
    } catch (error) {
      console.error('Cache delete error:', error);
    }
  }

  async getJSON<T>(key: string): Promise<T | null> {
    try {
      const value = await this.get(key);
      return value ? JSON.parse(value) : null;
    } catch (error) {
      console.error('Cache getJSON error:', error);
      return null;
    }
  }

  async setJSON<T>(key: string, value: T, ttl?: number): Promise<void> {
    try {
      await this.set(key, JSON.stringify(value), ttl);
    } catch (error) {
      console.error('Cache setJSON error:', error);
    }
  }

  // 缓存模式：先检查缓存，如果不存在则执行函数并缓存结果
  async getOrSet<T>(
    key: string,
    factory: () => Promise<T>,
    ttl?: number
  ): Promise<T> {
    const cached = await this.getJSON<T>(key);
    if (cached !== null) {
      return cached;
    }

    const value = await factory();
    await this.setJSON(key, value, ttl);
    return value;
  }

  // 批量操作
  async mget(keys: string[]): Promise<(string | null)[]> {
    return Promise.all(keys.map(key => this.get(key)));
  }

  async mset(entries: Array<{ key: string; value: string; ttl?: number }>): Promise<void> {
    await Promise.all(
      entries.map(entry => this.set(entry.key, entry.value, entry.ttl))
    );
  }

  // 缓存失效模式
  async invalidatePattern(pattern: string): Promise<void> {
```

```typescript
import type { ErrorHandler } from 'hono';
import type { Context } from '../types/common';

export const errorHandler: ErrorHandler = (err, c) => {
  console.error('Application error:', err);

  // 开发环境返回详细错误信息
  const isDevelopment = c.env?.ENVIRONMENT === 'development';

  if (err instanceof ValidationError) {
    return c.json({
      error: 'Validation Error',
      message: err.message,
      details: isDevelopment ? err.details : undefined,
    }, 400);
  }

  if (err instanceof AuthenticationError) {
    return c.json({
      error: 'Authentication Error',
      message: err.message,
    }, 401);
  }

  if (err instanceof AuthorizationError) {
    return c.json({
      error: 'Authorization Error',
      message: err.message,
    }, 403);
  }

  if (err instanceof NotFoundError) {
    return c.json({
      error: 'Not Found',
      message: err.message,
    }, 404);
  }

  if (err instanceof RateLimitError) {
    return c.json({
      error: 'Rate Limit Exceeded',
      message: err.message,
      retryAfter: err.retryAfter,
    }, 429);
  }

  // 默认服务器错误
  return c.json({
    error: 'Internal Server Error',
    message: isDevelopment ? err.message : 'Something went wrong',

  }, 500);
```

```typescript
// src/utils/databaseOptimization.ts
export class DatabaseOptimization {
  // 批量插入优化
  static async batchInsert<T>(
    db: D1Database,
    table: string,
    records: T[],
    batchSize: number = 100
  ): Promise<void> {
    for (let i = 0; i < records.length; i += batchSize) {
      const batch = records.slice(i, i + batchSize);
      const placeholders = batch.map(() => '(?, ?, ?)').join(', ');
      const values = batch.flatMap(record => Object.values(record));
      
      const stmt = db.prepare(`INSERT INTO ${table} VALUES ${placeholders}`);
      await stmt.bind(...values).run();
    }
  }

  // 分页查询优化
  static buildPaginationQuery(
    baseQuery: string,
    page: number,
    limit: number,
    orderBy: string = 'created_at DESC'
  ): { query: string; offset: number } {
    const offset = (page - 1) * limit;
    const query = `${baseQuery} ORDER BY ${orderBy} LIMIT ${limit} OFFSET ${offset}`;
    return { query, offset };
  }

  // 索引建议
  static readonly INDEX_RECOMMENDATIONS = {
    users: ['email', 'status', 'created_at'],
    contents: ['user_id', 'type', 'status', 'created_at'],
    templates: ['category', 'is_public', 'created_by'],
  };
}
```

```typescript
      message,
      timestamp: new Date().toISOString(),
      context,
    };

    // 在生产环境中，可以发送到外部日志服务

  }

    this.log(LogLevel.DEBUG, message, context);
  }

  info(message: string, context?: any): void {
    this.log(LogLevel.INFO, message, context);
  }

  warn(message: string, context?: any): void {
    this.log(LogLevel.WARN, message, context);
  }

  error(message: string, context?: any): void {
    this.log(LogLevel.ERROR, message, context);
  }

  // 请求日志中间件
  static requestLogger() {
    return async (c: any, next: any) => {
      const start = Date.now();
      const requestId = crypto.randomUUID();
      
      c.set('requestId', requestId);
      
      Logger.getInstance().info('Request started', {
        requestId,
        method: c.req.method,
        url: c.req.url,
        userAgent: c.req.header('User-Agent'),
      });

      await next();

      const duration = Date.now() - start;
      Logger.getInstance().info('Request completed', {
        requestId,
        status: c.res.status,
        duration,
      });
    };
  }
```

```typescript
export const sanitizeInput = createMiddleware<Context>(async (c, next) => {
  const contentType = c.req.header('Content-Type');
  
  if (contentType?.includes('application/json')) {
    try {
      const body = await c.req.json();
      const sanitized = sanitizeObject(body);
      c.req.json = () => Promise.resolve(sanitized);
    } catch (error) {
      // 忽略JSON解析错误，让后续处理
    }
  }
  
  await next();
});

function sanitizeObject(obj: any): any {
  if (typeof obj === 'string') {
    return obj.replace(/<script[^>]*>.*?<\/script>/gi, '')
             .replace(/<[^>]*>/g, '')
             .trim();
  }
  
  if (Array.isArray(obj)) {
    return obj.map(sanitizeObject);
  }
  
  if (obj && typeof obj === 'object') {
    const sanitized: any = {};
    for (const [key, value] of Object.entries(obj)) {
      sanitized[key] = sanitizeObject(value);
    }
    return sanitized;
  }
  
  return obj;
}
```

```typescript
// src/utils/encryption.ts
export class EncryptionService {
  private key: string;

  constructor(key: string) {
    this.key = key;
  }

  async encrypt(data: string): Promise<string> {
    const encoder = new TextEncoder();
    const keyData = encoder.encode(this.key);
    const dataBuffer = encoder.encode(data);
    
    const cryptoKey = await crypto.subtle.importKey(
      'raw',
      keyData,
      { name: 'AES-GCM' },
      false,
      ['encrypt']
    );
    
    const iv = crypto.getRandomValues(new Uint8Array(12));
    const encrypted = await crypto.subtle.encrypt(
      { name: 'AES-GCM', iv },
      cryptoKey,
      dataBuffer
    );
    
    const result = new Uint8Array(iv.length + encrypted.byteLength);
    result.set(iv);
    result.set(new Uint8Array(encrypted), iv.length);
    
    return btoa(String.fromCharCode(...result));
  }

  async decrypt(encryptedData: string): Promise<string> {
    const data = new Uint8Array(
      atob(encryptedData).split('').map(char => char.charCodeAt(0))
    );
    
    const iv = data.slice(0, 12);
    const encrypted = data.slice(12);
    
    const encoder = new TextEncoder();
    const keyData = encoder.encode(this.key);
    
    const cryptoKey = await crypto.subtle.importKey(
      'raw',
      keyData,
      { name: 'AES-GCM' },
      false,
      ['decrypt']
    );
    
    const decrypted = await crypto.subtle.decrypt(
      { name: 'AES-GCM', iv },
      cryptoKey,
      encrypted
    );
    
    return new TextDecoder().decode(decrypted);
  }

  // 密码哈希（使用Web Crypto API）
  async hashPassword(password: string, salt?: string): Promise<string> {
    const encoder = new TextEncoder();
    const saltBytes = salt ? encoder.encode(salt) : crypto.getRandomValues(new Uint8Array(16));
    const passwordBytes = encoder.encode(password);
    
    const combined = new Uint8Array(passwordBytes.length + saltBytes.length);
    combined.set(passwordBytes);
    combined.set(saltBytes, passwordBytes.length);
    
    const hash = await crypto.subtle.digest('SHA-256', combined);
    const hashArray = new Uint8Array(hash);
    
    const result = new Uint8Array(saltBytes.length + hashArray.length);
    result.set(saltBytes);
    result.set(hashArray, saltBytes.length);
    
    return btoa(String.fromCharCode(...result));
  }
}
```
