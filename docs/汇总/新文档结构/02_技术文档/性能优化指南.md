// 性能基准测试套件
export class PerformanceBenchmark {
  constructor(baseUrl, concurrency = 100) {
    this.baseUrl = baseUrl;
    this.concurrency = concurrency;
    this.results = [];
  }

  async runBenchmark() {
    console.log('🚀 Starting performance benchmark...');
    
    const tests = [
      { name: 'API Health Check', endpoint: '/api/health', method: 'GET' },
      { name: 'User Authentication', endpoint: '/api/auth/login', method: 'POST' },
      { name: 'Content Rewrite', endpoint: '/api/content/rewrite', method: 'POST' },
      { name: 'Template Render', endpoint: '/api/templates/render', method: 'POST' },
      { name: 'User Profile', endpoint: '/api/user/profile', method: 'GET' }
    ];
    
    for (const test of tests) {
      const result = await this.benchmarkEndpoint(test);
      this.results.push(result);
    }
    
    return this.generateReport();
  }

  async benchmarkEndpoint(test) {
    const startTime = Date.now();
    const promises = [];
    const results = {
      name: test.name,
      endpoint: test.endpoint,
      method: test.method,
      concurrency: this.concurrency,
      responses: [],
      errors: []
    };
    
    // 创建并发请求
    for (let i = 0; i < this.concurrency; i++) {
      promises.push(this.makeRequest(test).catch(error => ({ error })));
    }
    
    // 等待所有请求完成
    const responses = await Promise.all(promises);
    
    // 分析结果
    for (const response of responses) {
      if (response.error) {
        results.errors.push(response.error);
      } else {
        results.responses.push(response);
      }
    }
    
    const totalTime = Date.now() - startTime;
    const responseTimes = results.responses.map(r => r.responseTime);
    
    results.statistics = {
      totalTime,
      successCount: results.responses.length,
      errorCount: results.errors.length,
      successRate: (results.responses.length / this.concurrency * 100).toFixed(2) + '%',
      avgResponseTime: this.average(responseTimes),
      minResponseTime: Math.min(...responseTimes),
      maxResponseTime: Math.max(...responseTimes),
      p50: this.percentile(responseTimes, 50),
      p95: this.percentile(responseTimes, 95),
      p99: this.percentile(responseTimes, 99),
      rps: (results.responses.length / (totalTime / 1000)).toFixed(2)
    };
    
    return results;
  }

  async makeRequest(test) {
    const startTime = Date.now();
    
    try {
      const response = await fetch(`${this.baseUrl}${test.endpoint}`, {
        method: test.method,
        headers: {
          'Content-Type': 'application/json',
          'Authorization': 'Bearer test-token'
        },
        body: test.method === 'POST' ? JSON.stringify(this.getTestData(test.endpoint)) : undefined
      });
      
      const responseTime = Date.now() - startTime;
      const body = await response.text();
      
      return {
        status: response.status,
        responseTime,
        size: body.length,
        headers: Object.fromEntries(response.headers.entries())
      };
    } catch (error) {
      return {
        error: error.message,
        responseTime: Date.now() - startTime
      };
    }
  }

  getTestData(endpoint) {
    const testData = {
      '/api/auth/login': {
        username: 'test@example.com',
        password: 'testpassword'
      },
      '/api/content/rewrite': {
        content: 'This is a test content for rewriting.',
        style: 'professional',
        tone: 'formal'
      },
      '/api/templates/render': {
        templateId: 'test-template',
        variables: { name: 'Test User', date: new Date().toISOString() }
      }
    };
    
    return testData[endpoint] || {};
  }

  average(arr) {
    return arr.length ? (arr.reduce((a, b) => a + b, 0) / arr.length).toFixed(2) : 0;
  }

  percentile(arr, p) {
    if (!arr.length) return 0;
    const sorted = arr.sort((a, b) => a - b);
    const index = Math.ceil((p / 100) * sorted.length) - 1;
    return sorted[index];
  }

  generateReport() {
    return {
      summary: {
        totalTests: this.results.length,
        totalRequests: this.results.reduce((sum, r) => sum + r.concurrency, 0),
        overallSuccessRate: this.calculateOverallSuccessRate(),
        avgResponseTime: this.calculateOverallAvgResponseTime()
      },
      details: this.results
    };
  }

  calculateOverallSuccessRate() {
    const totalRequests = this.results.reduce((sum, r) => sum + r.concurrency, 0);
    const totalSuccess = this.results.reduce((sum, r) => sum + r.statistics.successCount, 0);
    return ((totalSuccess / totalRequests) * 100).toFixed(2) + '%';
  }

  calculateOverallAvgResponseTime() {
    const allResponseTimes = this.results.flatMap(r => r.responses.map(res => res.responseTime));
    return this.average(allResponseTimes);
  }
}

import { useState, useRef, useEffect } from 'react';

function LazyImage({ src, alt, placeholder, className }) {
  const [isLoaded, setIsLoaded] = useState(false);
  const [isInView, setIsInView] = useState(false);
  const imgRef = useRef();

  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsInView(true);
          observer.disconnect();
        }
      },
      { threshold: 0.1 }
    );

    if (imgRef.current) {
      observer.observe(imgRef.current);
    }

    return () => observer.disconnect();
  }, []);

  return (
    <div ref={imgRef} className={className}>
      {isInView && (
        <img
          src={src}
          alt={alt}
          onLoad={() => setIsLoaded(true)}
          style={{
            opacity: isLoaded ? 1 : 0,
            transition: 'opacity 0.3s ease-in-out'
          }}
        />
      )}
      {!isLoaded && isInView && (
        <div className="placeholder">{placeholder}</div>
      )}
    </div>
  );
}

// Service Worker缓存策略
self.addEventListener('fetch', (event) => {
  const { request } = event;
  const url = new URL(request.url);
  
  // API请求缓存策略
  if (url.pathname.startsWith('/api/')) {
    event.respondWith(
      caches.open('api-cache').then(cache => {
        return cache.match(request).then(response => {
          if (response) {
            // 缓存命中，后台更新
            fetch(request).then(fetchResponse => {
              cache.put(request, fetchResponse.clone());
            });
            return response;
          }
          
          // 缓存未命中，网络请求
          return fetch(request).then(fetchResponse => {
            cache.put(request, fetchResponse.clone());
            return fetchResponse;
          });
        });
      })
    );
  }
  
  // 静态资源缓存策略
  if (url.pathname.match(/\.(js|css|png|jpg|jpeg|gif|svg)$/)) {
    event.respondWith(
      caches.match(request).then(response => {
        return response || fetch(request).then(fetchResponse => {
          const cache = caches.open('static-cache');
          cache.then(c => c.put(request, fetchResponse.clone()));
          return fetchResponse;
        });
      })
    );
  }
});

// 数据库连接池优化
export class DatabasePool {
  constructor(config) {
    this.config = config;
    this.connections = new Map();
    this.maxConnections = config.maxConnections || 10;
    this.connectionTimeout = config.connectionTimeout || 30000;
  }

  async getConnection() {
    const workerId = this.getWorkerId();
    
    if (!this.connections.has(workerId)) {
      const connection = await this.createConnection();
      this.connections.set(workerId, {
        connection,
        lastUsed: Date.now(),
        inUse: false
      });
    }
    
    const conn = this.connections.get(workerId);
    conn.lastUsed = Date.now();
    conn.inUse = true;
    
    return conn.connection;
  }

  releaseConnection(workerId) {
    const conn = this.connections.get(workerId);
    if (conn) {
      conn.inUse = false;
    }
  }

  // 清理过期连接
  cleanup() {
    const now = Date.now();
    for (const [workerId, conn] of this.connections) {
      if (!conn.inUse && (now - conn.lastUsed) > this.connectionTimeout) {
        this.connections.delete(workerId);
      }
    }
  }
}

export class OptimizedQueries {
  constructor(db) {
    this.db = db;
    this.queryCache = new Map();
  }

  // 批量查询
  async batchSelect(table, ids, fields = '*') {
    if (!ids.length) return [];
    
    const placeholders = ids.map(() => '?').join(',');
    const query = `SELECT ${fields} FROM ${table} WHERE id IN (${placeholders})`;
    
    return await this.db.prepare(query).bind(...ids).all();
  }

  // 分页查询优化
  async paginatedSelect(table, options = {}) {
    const {
      page = 1,
      limit = 20,
      orderBy = 'created_at',
      orderDirection = 'DESC',
      where = '',
      params = []
    } = options;
    
    const offset = (page - 1) * limit;
    
    // 使用LIMIT和OFFSET进行分页
    const query = `
      SELECT * FROM ${table}
      ${where ? `WHERE ${where}` : ''}
      ORDER BY ${orderBy} ${orderDirection}
      LIMIT ? OFFSET ?
    `;
    
    const countQuery = `
      SELECT COUNT(*) as total FROM ${table}
      ${where ? `WHERE ${where}` : ''}
    `;
    
    const [data, countResult] = await Promise.all([
      this.db.prepare(query).bind(...params, limit, offset).all(),
      this.db.prepare(countQuery).bind(...params).first()
    ]);
    
    return {
      data,
      pagination: {
        page,
        limit,
        total: countResult.total,
        totalPages: Math.ceil(countResult.total / limit)
      }
    };
  }

  // 查询结果缓存
  async cachedQuery(key, query, params = [], ttl = 300) {
    const cached = this.queryCache.get(key);
    
    if (cached && (Date.now() - cached.timestamp) < ttl * 1000) {
      return cached.data;
    }
    
    const data = await this.db.prepare(query).bind(...params).all();
    
    this.queryCache.set(key, {
      data,
      timestamp: Date.now()
    });
    
    return data;
  }

  // 事务优化
  async optimizedTransaction(operations) {
    const transaction = this.db.transaction(() => {
      for (const operation of operations) {
        operation();
      }
    });
    
    return transaction();
  }
}

// 多层缓存策略
export class MultiLevelCache {
  constructor(config) {
    this.memoryCache = new Map();
    this.kvCache = config.kvCache;
    this.memoryTTL = config.memoryTTL || 60; // 1分钟
    this.kvTTL = config.kvTTL || 3600; // 1小时
    this.maxMemorySize = config.maxMemorySize || 1000;
  }

  async get(key) {
    // L1: 内存缓存
    const memoryResult = this.memoryCache.get(key);
    if (memoryResult && !this.isExpired(memoryResult)) {
      return memoryResult.data;
    }
    
    // L2: KV缓存
    try {
      const kvResult = await this.kvCache.get(key, 'json');
      if (kvResult) {
        // 回填内存缓存
        this.setMemoryCache(key, kvResult);
        return kvResult;
      }
    } catch (error) {
      console.warn('KV cache error:', error);
    }
    
    return null;
  }

  async set(key, data, options = {}) {
    const memoryTTL = options.memoryTTL || this.memoryTTL;
    const kvTTL = options.kvTTL || this.kvTTL;
    
    // 设置内存缓存
    this.setMemoryCache(key, data, memoryTTL);
    
    // 设置KV缓存
    try {
      await this.kvCache.put(key, JSON.stringify(data), {
        expirationTtl: kvTTL
      });
    } catch (error) {
      console.warn('KV cache set error:', error);
    }
  }

  setMemoryCache(key, data, ttl = this.memoryTTL) {
    // 检查内存缓存大小
    if (this.memoryCache.size >= this.maxMemorySize) {
      this.evictOldest();
    }
    
    this.memoryCache.set(key, {
      data,
      timestamp: Date.now(),
      ttl: ttl * 1000
    });
  }

  evictOldest() {
    let oldestKey = null;
    let oldestTime = Date.now();
    
    for (const [key, value] of this.memoryCache) {
      if (value.timestamp < oldestTime) {
        oldestTime = value.timestamp;
        oldestKey = key;
      }
    }
    
    if (oldestKey) {
      this.memoryCache.delete(oldestKey);
    }
  }

  isExpired(cacheItem) {
    return Date.now() - cacheItem.timestamp > cacheItem.ttl;
  }

  // 缓存预热
  async warmup(keys) {
    const promises = keys.map(async (key) => {
      try {
        const data = await this.generateData(key);
        await this.set(key, data);
      } catch (error) {
        console.warn(`Cache warmup failed for key ${key}:`, error);
      }
    });
    
    await Promise.allSettled(promises);
  }

  // 缓存统计
  getStats() {
    return {
      memorySize: this.memoryCache.size,
      memoryHitRate: this.calculateHitRate('memory'),
      kvHitRate: this.calculateHitRate('kv')
    };
  }
}

export class SmartCache {
  constructor(cache) {
    this.cache = cache;
    this.accessPatterns = new Map();
    this.hitRates = new Map();
  }

  async get(key) {
    this.recordAccess(key);
    
    const result = await this.cache.get(key);
    
    if (result) {
      this.recordHit(key);
    } else {
      this.recordMiss(key);
    }
    
    return result;
  }

  async set(key, data) {
    const ttl = this.calculateOptimalTTL(key);
    await this.cache.set(key, data, { kvTTL: ttl });
  }

  recordAccess(key) {
    const pattern = this.accessPatterns.get(key) || {
      count: 0,
      lastAccess: 0,
      frequency: 0
    };
    
    const now = Date.now();
    const timeDiff = now - pattern.lastAccess;
    
    pattern.count++;
    pattern.lastAccess = now;
    pattern.frequency = pattern.count / (timeDiff || 1);
    
    this.accessPatterns.set(key, pattern);
  }

  calculateOptimalTTL(key) {
    const pattern = this.accessPatterns.get(key);
    if (!pattern) return 3600; // 默认1小时
    
    // 根据访问频率调整TTL
    if (pattern.frequency > 0.1) {
      return 7200; // 高频访问，2小时
    } else if (pattern.frequency > 0.01) {
      return 3600; // 中频访问，1小时
    } else {
      return 1800; // 低频访问，30分钟
    }
  }
}

// 请求合并和批处理
export class RequestBatcher {
  constructor(options = {}) {
    this.batchSize = options.batchSize || 10;
    this.batchTimeout = options.batchTimeout || 100;
    this.batches = new Map();
  }

  async batch(key, request) {
    return new Promise((resolve, reject) => {
      if (!this.batches.has(key)) {
        this.batches.set(key, {
          requests: [],
          timer: null
        });
      }
      
      const batch = this.batches.get(key);
      batch.requests.push({ request, resolve, reject });
      
      // 如果达到批次大小，立即执行
      if (batch.requests.length >= this.batchSize) {
        this.executeBatch(key);
      } else if (!batch.timer) {
        // 设置超时执行
        batch.timer = setTimeout(() => {
          this.executeBatch(key);
        }, this.batchTimeout);
      }
    });
  }

  async executeBatch(key) {
    const batch = this.batches.get(key);
    if (!batch || !batch.requests.length) return;
    
    // 清除定时器
    if (batch.timer) {
      clearTimeout(batch.timer);
    }
    
    // 提取请求
    const requests = batch.requests.splice(0);
    this.batches.delete(key);
    
    try {
      // 执行批量请求
      const results = await this.processBatch(key, requests.map(r => r.request));
      
      // 返回结果
      requests.forEach((req, index) => {
        req.resolve(results[index]);
      });
    } catch (error) {
      // 处理错误
      requests.forEach(req => {
        req.reject(error);
      });
    }
  }

  async processBatch(key, requests) {
    // 根据key类型处理不同的批量请求
    switch (key) {
      case 'user-profiles':
        return this.batchUserProfiles(requests);
      case 'content-analysis':
        return this.batchContentAnalysis(requests);
      default:
        throw new Error(`Unknown batch key: ${key}`);
    }
  }

  async batchUserProfiles(requests) {
    const userIds = requests.map(req => req.userId);
    const profiles = await this.db.prepare(`
      SELECT * FROM users WHERE id IN (${userIds.map(() => '?').join(',')})
    `).bind(...userIds).all();
    
    // 按请求顺序返回结果
    return requests.map(req => 
      profiles.find(profile => profile.id === req.userId)
    );
  }
}

// 并发控制
export class ConcurrencyLimiter {
  constructor(maxConcurrent = 10) {
    this.maxConcurrent = maxConcurrent;
    this.running = 0;
    this.queue = [];
  }

  async execute(fn) {
    return new Promise((resolve, reject) => {
      this.queue.push({ fn, resolve, reject });
      this.process();
    });
  }

  async process() {
    if (this.running >= this.maxConcurrent || this.queue.length === 0) {
      return;
    }
    
    this.running++;
    const { fn, resolve, reject } = this.queue.shift();
    
    try {
      const result = await fn();
      resolve(result);
    } catch (error) {
      reject(error);
    } finally {
      this.running--;
      this.process(); // 处理下一个任务
    }

// AI模型调用优化
export class AIModelOptimizer {
  constructor(config) {
    this.config = config;
    this.requestQueue = [];
    this.rateLimiter = new RateLimiter(config.rateLimit);
    this.cache = new Map();
  }

  async optimizeRequest(prompt, options = {}) {
    // 1. 检查缓存
    const cacheKey = this.generateCacheKey(prompt, options);
    const cached = this.cache.get(cacheKey);
    
    if (cached && !this.isCacheExpired(cached)) {
      return cached.result;
    }
    
    // 2. 优化prompt
    const optimizedPrompt = this.optimizePrompt(prompt);
    
    // 3. 批量处理
    if (options.allowBatching) {
      return this.batchRequest(optimizedPrompt, options);
    }
    
    // 4. 单独处理
    return this.singleRequest(optimizedPrompt, options, cacheKey);
  }

  optimizePrompt(prompt) {
    // 移除多余空白
    let optimized = prompt.replace(/\s+/g, ' ').trim();
    
    // 压缩常见模式
    const patterns = [
      { pattern: /请帮我/g, replacement: '请' },
      { pattern: /能否帮助我/g, replacement: '请' },
      { pattern: /非常感谢/g, replacement: '谢谢' }
    ];
    
    patterns.forEach(({ pattern, replacement }) => {
      optimized = optimized.replace(pattern, replacement);
    });
    
    return optimized;
  }

  async batchRequest(prompt, options) {
    return new Promise((resolve, reject) => {
      this.requestQueue.push({ prompt, options, resolve, reject });
      
      // 延迟执行批处理
      setTimeout(() => this.processBatch(), 100);
    });
  }

  async processBatch() {
    if (this.requestQueue.length === 0) return;
    
    const batch = this.requestQueue.splice(0, 5); // 最多5个请求一批
    
    try {
      const results = await this.callAIModel(batch);
      
      batch.forEach((req, index) => {
        const result = results[index];
        const cacheKey = this.generateCacheKey(req.prompt, req.options);
        
        // 缓存结果
        this.cache.set(cacheKey, {
          result,
          timestamp: Date.now()
        });
        
        req.resolve(result);
      });
    } catch (error) {
      batch.forEach(req => req.reject(error));
    }
  }

  async singleRequest(prompt, options, cacheKey) {
    // 速率限制
    await this.rateLimiter.waitForSlot();
    
    try {
      const result = await this.callAIModel([{ prompt, options }]);
      
      // 缓存结果
      this.cache.set(cacheKey, {
        result: result[0],
        timestamp: Date.now()
      });
      
      return result[0];
    } catch (error) {
      throw error;
    }
  }

  generateCacheKey(prompt, options) {
    const key = JSON.stringify({ prompt, options });
    return this.hash(key);
  }

  hash(str) {
    let hash = 0;
    for (let i = 0; i < str.length; i++) {
      const char = str.charCodeAt(i);
      hash = ((hash << 5) - hash) + char;
      hash = hash & hash; // 转换为32位整数
    }
    return hash.toString();
  }

  isCacheExpired(cached) {
    const ttl = 30 * 60 * 1000; // 30分钟
    return Date.now() - cached.timestamp > ttl;
  }
}

// 智能重试机制
export class SmartRetry {
  constructor(options = {}) {
    this.maxRetries = options.maxRetries || 3;
    this.baseDelay = options.baseDelay || 1000;
    this.maxDelay = options.maxDelay || 10000;
    this.backoffFactor = options.backoffFactor || 2;
  }

  async execute(fn, context = {}) {
    let lastError;
    
    for (let attempt = 0; attempt <= this.maxRetries; attempt++) {
      try {
        return await fn();
      } catch (error) {
        lastError = error;
        
        // 检查是否应该重试
        if (!this.shouldRetry(error, attempt)) {
          throw error;
        }
        
        // 计算延迟时间
        const delay = this.calculateDelay(attempt);
        
        // 记录重试
        console.warn(`Retry attempt ${attempt + 1}/${this.maxRetries} after ${delay}ms:`, error.message);
        
        // 等待后重试
        await this.sleep(delay);
      }
    }
    
    throw lastError;
  }

  shouldRetry(error, attempt) {
    if (attempt >= this.maxRetries) return false;
    
    // 不重试的错误类型
    const nonRetryableErrors = [
      'INVALID_API_KEY',
      'QUOTA_EXCEEDED',
      'INVALID_REQUEST'
    ];
    
    if (nonRetryableErrors.includes(error.code)) {
      return false;
    }
    
    // HTTP状态码判断
    if (error.status) {
      // 4xx错误通常不应重试（除了429）
      if (error.status >= 400 && error.status < 500 && error.status !== 429) {
        return false;
      }
    }
    
    return true;
  }

  calculateDelay(attempt) {
    const delay = this.baseDelay * Math.pow(this.backoffFactor, attempt);
    const jitter = Math.random() * 0.1 * delay; // 添加10%的随机抖动
    return Math.min(delay + jitter, this.maxDelay);
  }

  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

// 流式响应处理
export class StreamProcessor {
  constructor() {
    this.bufferSize = 1024;
    this.processors = new Map();
  }

  async processStream(stream, processorType, options = {}) {
    const processor = this.processors.get(processorType);
    if (!processor) {
      throw new Error(`Unknown processor type: ${processorType}`);
    }
    
    const reader = stream.getReader();
    const decoder = new TextDecoder();
    let buffer = '';
    let result = '';
    
    try {
      while (true) {
        const { done, value } = await reader.read();
        
        if (done) break;
        
        // 解码数据
        const chunk = decoder.decode(value, { stream: true });
        buffer += chunk;
        
        // 处理完整的行
        const lines = buffer.split('\n');
        buffer = lines.pop() || ''; // 保留不完整的行
        
        for (const line of lines) {
          if (line.trim()) {
            const processed = await processor(line, options);
            result += processed;
            
            // 实时返回处理结果
            if (options.onChunk) {
              options.onChunk(processed);
            }
          }
        }
      }
      
      // 处理剩余的buffer
      if (buffer.trim()) {
        const processed = await processor(buffer, options);
        result += processed;
        
        if (options.onChunk) {
          options.onChunk(processed);
        }
      }
      
      return result;
    } finally {
      reader.releaseLock();
    }
  }

  registerProcessor(type, processor) {
    this.processors.set(type, processor);
  }
}

export class PerformanceMonitor {
  constructor(config) {
    this.config = config;
    this.metrics = new Map();
    this.alerts = [];
    this.thresholds = config.thresholds || {};
  }

  // 记录指标
  recordMetric(name, value, tags = {}) {
    const timestamp = Date.now();
    const key = `${name}:${JSON.stringify(tags)}`;
    
    if (!this.metrics.has(key)) {
      this.metrics.set(key, []);
    }
    
    const values = this.metrics.get(key);
    values.push({ value, timestamp });
    
    // 保持最近1小时的数据
    const oneHourAgo = timestamp - 3600000;
    this.metrics.set(key, values.filter(v => v.timestamp > oneHourAgo));
    
    // 检查阈值
    this.checkThreshold(name, value, tags);
  }

  // 检查阈值告警
  checkThreshold(name, value, tags) {
    const threshold = this.thresholds[name];
    if (!threshold) return;
    
    if (value > threshold.critical) {
      this.triggerAlert('critical', name, value, threshold.critical, tags);
    } else if (value > threshold.warning) {
      this.triggerAlert('warning', name, value, threshold.warning, tags);
    }
  }

  triggerAlert(level, metric, value, threshold, tags) {
    const alert = {
      id: generateUUID(),
      level,
      metric,
      value,
      threshold,
      tags,
      timestamp: Date.now()
    };
    
    this.alerts.push(alert);
    
    // 发送告警通知
    this.sendAlert(alert);
  }

  async sendAlert(alert) {
    if (this.config.webhookUrl) {
      try {
        await fetch(this.config.webhookUrl, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            text: `🚨 Performance Alert: ${alert.metric} = ${alert.value} (threshold: ${alert.threshold})`,
            level: alert.level,
            details: alert
          })
        });
      } catch (error) {
        console.error('Failed to send alert:', error);
      }
    }
  }

  // 获取统计信息
  getStats(name, timeRange = 3600000) { // 默认1小时
    const now = Date.now();
    const startTime = now - timeRange;
    
    const allValues = [];
    
    for (const [key, values] of this.metrics) {
      if (key.startsWith(name + ':')) {
        const recentValues = values
          .filter(v => v.timestamp >= startTime)
          .map(v => v.value);
        allValues.push(...recentValues);
      }
    }
    
    if (allValues.length === 0) {
      return null;
    }
    
    allValues.sort((a, b) => a - b);
    
    return {
      count: allValues.length,
      min: allValues[0],
      max: allValues[allValues.length - 1],
      avg: allValues.reduce((sum, v) => sum + v, 0) / allValues.length,
      p50: allValues[Math.floor(allValues.length * 0.5)],
      p95: allValues[Math.floor(allValues.length * 0.95)],
      p99: allValues[Math.floor(allValues.length * 0.99)]
    };
  }
}

// 异常检测器
class AnomalyDetector {
  constructor() {
    this.baselines = new Map();
    this.sensitivity = 2; // 标准差倍数
  }

  detect(metric, stats) {
    const baseline = this.baselines.get(metric);
    
    if (!baseline) {
      // 建立基线
      this.baselines.set(metric, {
        mean: stats.avg,
        stdDev: this.calculateStdDev(stats),
        samples: 1
      });
      return null;
    }
    
    // 更新基线
    this.updateBaseline(metric, stats);
    
    // 检测异常
    const threshold = baseline.mean + (this.sensitivity * baseline.stdDev);
    
    if (stats.avg > threshold) {
      return {
        metric,
        type: 'spike',
        severity: stats.avg > threshold * 1.5 ? 'critical' : 'warning',
        value: stats.avg,
        baseline: baseline.mean,
        threshold
      };
    }
    
    return null;
  }

  updateBaseline(metric, stats) {
    const baseline = this.baselines.get(metric);
    const alpha = 0.1; // 学习率
    
    baseline.mean = baseline.mean * (1 - alpha) + stats.avg * alpha;
    baseline.samples++;
  }

  calculateStdDev(stats) {
    // 简化的标准差计算
    return (stats.max - stats.min) / 4;
  }
}
