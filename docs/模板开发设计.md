## 特殊功能实现

### 1. 目录生成

```javascript
// 自动生成文章目录
function generateTOC(content) {
  const headings = content.match(/<h[1-6][^>]*>.*?<\/h[1-6]>/gi) || [];
  
  if (headings.length === 0) return '';
  
  let toc = '<nav class="table-of-contents"><h3>目录</h3><ul>';
  
  headings.forEach((heading, index) => {
    const level = heading.match(/<h([1-6])/)[1];
    const text = heading.replace(/<[^>]*>/g, '');
    const id = `heading-${index}`;
    
    // 为标题添加ID
    content = content.replace(heading, heading.replace('>', ` id="${id}">`);
    
    toc += `<li class="toc-level-${level}"><a href="#${id}">${text}</a></li>`;
  });
  
  toc += '</ul></nav>';
  
  return { toc, content };
}
```

### 2. 错误处理

```javascript
// 渲染错误处理
render: function(content, title = '文档') {
  try {
    // 验证输入
    if (!content || typeof content !== 'string') {
      throw new Error('Invalid content provided');
    }
    
    // 处理特殊字符
    const safeTitle = this.escapeHtml(title);
    const safeContent = this.processContent(content);
    
    return this.generateHTML(safeContent, safeTitle);
    
  } catch (error) {
    console.error(`Template rendering error: ${error.message}`);
    return this.getFallbackHTML(title);
  }
},

// HTML转义
escapeHtml: function(text) {
  const div = document.createElement('div');
  div.textContent = text;
  return div.innerHTML;
},

// 降级HTML
getFallbackHTML: function(title) {
  return `
    <!DOCTYPE html>
    <html>
    <head><title>${title}</title></head>
    <body>
      <h1>渲染错误</h1>
      <p>模板渲染失败，请稍后重试。</p>
    </body>
    </html>
  `;
}

// templates/custom.js
export default {
  name: 'custom',
  displayName: '自定义模板',
  description: '根据特定需求定制的模板',
  version: '1.0.0',
  author: 'Your Name',
  
  // 模板配置
  config: {
    showTOC: true,
    enableCodeHighlight: true,
    theme: 'light'
  },
  
  render: function(content, title = '文档') {
    // 处理内容
    let processedContent = content;
    
    if (this.config.showTOC) {
      const { toc, content: contentWithIds } = this.generateTOC(processedContent);
      processedContent = toc + contentWithIds;
    }
    
    if (this.config.enableCodeHighlight) {
      processedContent = this.highlightCode(processedContent);
    }
    
    return this.generateHTML(processedContent, title);
  },
  
  generateHTML: function(content, title) {
    return `<!DOCTYPE html>
<html lang="zh-CN" data-theme="${this.config.theme}">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>${title}</title>
  <style>${this.getStyles()}</style>
</head>
<body>
  <div class="container">
    <header class="header">
      <h1 class="title">${title}</h1>
    </header>
    <main class="content">
      ${content}
    </main>
  </div>
  <script>${this.getScripts()}</script>
</body>
</html>`;
  },
  
  getStyles: function() {
    return `
      /* 自定义样式 */
      [data-theme="light"] {
        --bg-color: #ffffff;
        --text-color: #333333;
      }
      
      [data-theme="dark"] {
        --bg-color: #1a1a1a;
        --text-color: #ffffff;
      }
      
      body {
        background-color: var(--bg-color);
        color: var(--text-color);
        transition: background-color 0.3s, color 0.3s;
      }
      
      /* 更多自定义样式... */
    `;
  },
  
  getScripts: function() {
    return `
      // 主题切换功能
      function toggleTheme() {
        const html = document.documentElement;
        const currentTheme = html.getAttribute('data-theme');
        const newTheme = currentTheme === 'light' ? 'dark' : 'light';
        html.setAttribute('data-theme', newTheme);
        localStorage.setItem('theme', newTheme);
      }
      
      // 初始化主题
      const savedTheme = localStorage.getItem('theme') || 'light';
      document.documentElement.setAttribute('data-theme', savedTheme);
    `;
  },
  
  // 辅助方法
  generateTOC: function(content) {
    // TOC生成逻辑
    return { toc: '', content };
  },
  
  highlightCode: function(content) {
    // 代码高亮逻辑
    return content;
  }
};

// 验证模板结构
function validateTemplate(template) {
  const required = ['name', 'displayName', 'description', 'render'];
  
  for (const prop of required) {
    if (!template[prop]) {
      throw new Error(`Template missing required property: ${prop}`);
    }
  }
  
  if (typeof template.render !== 'function') {
    throw new Error('Template render must be a function');
  }
  
  // 测试渲染
  try {
    const testResult = template.render('<p>Test</p>', 'Test');
    if (!testResult || typeof testResult !== 'string') {
      throw new Error('Template render must return a string');
    }
  } catch (error) {
    throw new Error(`Template render test failed: ${error.message}`);
  }
  
  return true;
}

## 版本管理和兼容性

### 1. 模板版本控制

// 版本兼容性检查
function checkTemplateCompatibility(template, systemVersion) {
  const templateVersion = template.version || '1.0.0';
  const minVersion = template.minSystemVersion || '1.0.0';
  
  if (compareVersions(systemVersion, minVersion) < 0) {
    throw new Error(`Template requires system version ${minVersion} or higher`);
  }
  
  return true;
}

function compareVersions(a, b) {
  const aParts = a.split('.').map(Number);
  const bParts = b.split('.').map(Number);
  
  for (let i = 0; i < Math.max(aParts.length, bParts.length); i++) {
    const aPart = aParts[i] || 0;
    const bPart = bParts[i] || 0;
    
    if (aPart > bPart) return 1;
    if (aPart < bPart) return -1;
  }
  
  return 0;
}
```

// 模板迁移助手
class TemplateMigrator {
  static migrate(template, fromVersion, toVersion) {
    const migrations = {
      '1.0.0_to_1.1.0': this.migrateV1ToV1_1,
      '1.1.0_to_2.0.0': this.migrateV1_1ToV2
    };
    
    const migrationKey = `${fromVersion}_to_${toVersion}`;
    const migration = migrations[migrationKey];
    
    if (migration) {
      return migration(template);
    }
    
    return template;
  }
  
  static migrateV1ToV1_1(template) {
    // 添加新的配置选项
    if (!template.config) {
      template.config = {};
    }
    
    // 更新样式结构
    if (template.getStyles) {
      const oldGetStyles = template.getStyles;
      template.getStyles = function() {
        return this.wrapInMediaQueries(oldGetStyles.call(this));
      };
    }
    
    return template;
  }
}

模板系统是AI驱动内容代理的核心组件之一，通过合理的设计和开发，可以为用户提供丰富多样的内容展示效果。在开发模板时，需要注意：
